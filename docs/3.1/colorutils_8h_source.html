37 // fill_gradient - fill an array of colors with a smooth HSV gradient
38 // between two specified HSV colors.
39 // Since 'hue' is a value around a color wheel,
40 // there are always two ways to sweep from one hue
41 // to another.
42 // This function lets you specify which way you want
43 // the hue gradient to sweep around the color wheel:
44 // FORWARD_HUES: hue always goes clockwise
45 // BACKWARD_HUES: hue always goes counter-clockwise
46 // SHORTEST_HUES: hue goes whichever way is shortest
47 // LONGEST_HUES: hue goes whichever way is longest
48 // The default is SHORTEST_HUES, as this is nearly
49 // always what is wanted.
50 //
51 // fill_gradient can write the gradient colors EITHER
52 // (1) into an array of CRGBs (e.g., into leds[] array, or an RGB Palette)
53 // OR
54 // (2) into an array of CHSVs (e.g. an HSV Palette).
55 //
56 // In the case of writing into a CRGB array, the gradient is
57 // computed in HSV space, and then HSV values are converted to RGB
58 // as they're written into the RGB array.
59 
60 typedef enum { FORWARD_HUES, BACKWARD_HUES, SHORTEST_HUES, LONGEST_HUES } TGradientDirectionCode;

89 void fill_gradient( T* targetArray,
90  uint16_t startpos, CHSV startcolor,
91  uint16_t endpos, CHSV endcolor,
92  TGradientDirectionCode directionCode = SHORTEST_HUES )
93 {
94  // if the points are in the wrong order, straighten them
95  if( endpos < startpos ) {
96  uint16_t t = endpos;
97  CHSV tc = endcolor;
98  startpos = t;
99  startcolor = tc;
100  endcolor = startcolor;
101  endpos = startpos;
102  }
103 
104  // If we're fading toward black (val=0) or white (sat=0),
105  // then set the endhue to the starthue.
106  // This lets us ramp smoothly to black or white, regardless
107  // of what 'hue' was set in the endcolor (since it doesn't matter)
108  if( endcolor.value == 0 || endcolor.saturation == 0) {
109  endcolor.hue = startcolor.hue;
110  }
111 
112  // Similarly, if we're fading in from black (val=0) or white (sat=0)
113  // then set the starthue to the endhue.
114  // This lets us ramp smoothly up from black or white, regardless
115  // of what 'hue' was set in the startcolor (since it doesn't matter)
116  if( startcolor.value == 0 || startcolor.saturation == 0) {
117  startcolor.hue = endcolor.hue;
118  }
119 
120  saccum87 huedistance87;
121  saccum87 satdistance87;
122  saccum87 valdistance87;
123 
124  satdistance87 = (endcolor.sat - startcolor.sat) << 7;
125  valdistance87 = (endcolor.val - startcolor.val) << 7;
126 
127  uint8_t huedelta8 = endcolor.hue - startcolor.hue;
128 
129  if( directionCode == SHORTEST_HUES ) {
130  directionCode = FORWARD_HUES;
131  if( huedelta8 > 127) {
132  directionCode = BACKWARD_HUES;
133  }
134  }
135 
136  if( directionCode == LONGEST_HUES ) {
137  directionCode = FORWARD_HUES;
138  if( huedelta8 < 128) {
139  directionCode = BACKWARD_HUES;
140  }
141  }
142 
143  if( directionCode == FORWARD_HUES) {
144  huedistance87 = huedelta8 << 7;
145  }
146  else /* directionCode == BACKWARD_HUES */
147  {
148  huedistance87 = (uint8_t)(256 - huedelta8) << 7;
149  huedistance87 = -huedistance87;
150  }
151 
152  uint16_t pixeldistance = endpos - startpos;
153  int16_t divisor = pixeldistance ? pixeldistance : 1;
154 
155  saccum87 huedelta87 = huedistance87 / divisor;
156  saccum87 satdelta87 = satdistance87 / divisor;
157  saccum87 valdelta87 = valdistance87 / divisor;
158 
159  huedelta87 *= 2;
160  satdelta87 *= 2;
161  valdelta87 *= 2;
162 
163  accum88 hue88 = startcolor.hue << 8;
164  accum88 sat88 = startcolor.sat << 8;
165  accum88 val88 = startcolor.val << 8;
166  for( uint16_t i = startpos; i <= endpos; i++) {
167  targetArray[i] = CHSV( hue88 >> 8, sat88 >> 8, val88 >> 8);
168  hue88 += huedelta87;
169  sat88 += satdelta87;
170  val88 += valdelta87;
171  }
172 }

333 // Palettes
334 //
335 // RGB Palettes map an 8-bit value (0..255) to an RGB color.
336 //
337 // You can create any color palette you wish; a couple of starters
338 // are provided: Forest, Clouds, Lava, Ocean, Rainbow, and Rainbow Stripes.
339 //
340 // Palettes come in the traditional 256-entry variety, which take
341 // up 768 bytes of RAM, and lightweight 16-entry varieties. The 16-entry
342 // variety automatically interpolates between its entries to produce
343 // a full 256-element color map, but at a cost of only 48 bytes or RAM.
344 //
345 // Basic operation is like this: (example shows the 16-entry variety)
346 // 1. Declare your palette storage:
347 // CRGBPalette16 myPalette;
348 //
349 // 2. Fill myPalette with your own 16 colors, or with a preset color scheme.
350 // You can specify your 16 colors a variety of ways:
351 // CRGBPalette16 myPalette(
352 // CRGB::Black,
353 // CRGB::Black,
354 // CRGB::Red,
355 // CRGB::Yellow,
356 // CRGB::Green,
357 // CRGB::Blue,
358 // CRGB::Purple,
359 // CRGB::Black,
360 //
361 // 0x100000,
362 // 0x200000,
363 // 0x400000,
364 // 0x800000,
365 //
366 // CHSV( 30,255,255),
367 // CHSV( 50,255,255),
368 // CHSV( 70,255,255),
369 // CHSV( 90,255,255)
370 // );
371 //
372 // Or you can initiaize your palette with a preset color scheme:
373 // myPalette = RainbowStripesColors_p;
374 //
375 // 3. Any time you want to set a pixel to a color from your palette, use
376 // "ColorFromPalette(...)" as shown:
377 //
378 // uint8_t index = /* any value 0..255 */;
379 // leds[i] = ColorFromPalette( myPalette, index);
380 //
381 // Even though your palette has only 16 explicily defined entries, you
382 // can use an 'index' from 0..255. The 16 explicit palette entries will
383 // be spread evenly across the 0..255 range, and the intermedate values
384 // will be RGB-interpolated between adjacent explicit entries.
385 //
386 // It's easier to use than it sounds.
387 //

706  // Gradient palettes are loaded into CRGB16Palettes in such a way
707  // that, if possible, every color represented in the gradient palette
708  // is also represented in the CRGBPalette16.
709  // For example, consider a gradient palette that is all black except
710  // for a single, one-element-wide (1/256th!) spike of red in the middle:
711  // 0, 0,0,0
712  // 124, 0,0,0
713  // 125, 255,0,0 // one 1/256th-palette-wide red stripe
714  // 126, 0,0,0
715  // 255, 0,0,0
716  // A naive conversion of this 256-element palette to a 16-element palette
717  // might accidentally completely eliminate the red spike, rendering the
718  // palette completely black.
719  // However, the conversions provided here would attempt to include a
720  // the red stripe in the output, more-or-less as faithfully as possible.
721  // So in this case, the resulting CRGBPalette16 palette would have a red
722  // stripe in the middle which was 1/16th of a palette wide -- the
723  // narrowest possible in a CRGBPalette16.
724  // This means that the relative width of stripes in a CRGBPalette16
725  // will be, by definition, different from the widths in the gradient
726  // palette. This code attempts to preserve "all the colors", rather than
727  // the exact stripe widths at the expense of dropping some colors.

1046 // nblendPaletteTowardPalette:
1047 // Alter one palette by making it slightly more like
1048 // a 'target palette', used for palette cross-fades.
1049 //
1050 // It does this by comparing each of the R, G, and B channels
1051 // of each entry in the current palette to the corresponding
1052 // entry in the target palette and making small adjustments:
1053 // If the Red channel is too low, it will be increased.
1054 // If the Red channel is too high, it will be slightly reduced.
1055 // ... and likewise for Green and Blue channels.
1056 //
1057 // Additionally, there are two significant visual improvements
1058 // to this algorithm implemented here. First is this:
1059 // When increasing a channel, it is stepped up by ONE.
1060 // When decreasing a channel, it is stepped down by TWO.
1061 // Due to the way the eye perceives light, and the way colors
1062 // are represented in RGB, this produces a more uniform apparent
1063 // brightness when cross-fading between most palette colors.
1064 //
1065 // The second visual tweak is limiting the number of changes
1066 // that will be made to the palette at once. If all the palette
1067 // entries are changed at once, it can give a muddled appearance.
1068 // However, if only a few palette entries are changed at once,
1069 // you get a visually smoother transition: in the middle of the
1070 // cross-fade your current palette will actually contain some
1071 // colors from the old palette, a few blended colors, and some
1072 // colors from the new palette.
1073 // The maximum number of possible palette changes per call
1074 // is 48 (sixteen color entries time three channels each).
