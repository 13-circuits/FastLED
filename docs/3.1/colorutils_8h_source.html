80  // If we're fading toward black (val=0) or white (sat=0),
81  // then set the endhue to the starthue.
82  // This lets us ramp smoothly to black or white, regardless
83  // of what 'hue' was set in the endcolor (since it doesn't matter)
84  if( endcolor.value == 0 || endcolor.saturation == 0) {
85  endcolor.hue = startcolor.hue;
86  }
87 
88  // Similarly, if we're fading in from black (val=0) or white (sat=0)
89  // then set the starthue to the endhue.
90  // This lets us ramp smoothly up from black or white, regardless
91  // of what 'hue' was set in the startcolor (since it doesn't matter)
92  if( startcolor.value == 0 || startcolor.saturation == 0) {
93  startcolor.hue = endcolor.hue;
94  }

96  saccum87 huedistance87;
97  saccum87 satdistance87;
98  saccum87 valdistance87;
99 
100  satdistance87 = (endcolor.sat - startcolor.sat) << 7;
101  valdistance87 = (endcolor.val - startcolor.val) << 7;
102 
103  uint8_t huedelta8 = endcolor.hue - startcolor.hue;
104 
105  if( directionCode == SHORTEST_HUES ) {
106  directionCode = FORWARD_HUES;
107  if( huedelta8 > 127) {
108  directionCode = BACKWARD_HUES;
109  }
110  }
111 
112  if( directionCode == LONGEST_HUES ) {
113  directionCode = FORWARD_HUES;
114  if( huedelta8 < 128) {
115  directionCode = BACKWARD_HUES;
116  }
117  }
118 
119  if( directionCode == FORWARD_HUES) {
120  huedistance87 = huedelta8 << 7;
121  }
122  else /* directionCode == BACKWARD_HUES */
123  {
124  huedistance87 = (uint8_t)(256 - huedelta8) << 7;
125  huedistance87 = -huedistance87;
126  }
127 
128  uint16_t pixeldistance = endpos - startpos;
129  int16_t divisor = pixeldistance ? pixeldistance : 1;
130 
131  saccum87 huedelta87 = huedistance87 / divisor;
132  saccum87 satdelta87 = satdistance87 / divisor;
133  saccum87 valdelta87 = valdistance87 / divisor;
134 
135  huedelta87 *= 2;
136  satdelta87 *= 2;
137  valdelta87 *= 2;

143  targetArray[i] = CHSV( hue88 >> 8, sat88 >> 8, val88 >> 8);

309 // Palettes
310 //
311 // RGB Palettes map an 8-bit value (0..255) to an RGB color.
312 //
313 // You can create any color palette you wish; a couple of starters
314 // are provided: Forest, Clouds, Lava, Ocean, Rainbow, and Rainbow Stripes.
315 //
316 // Palettes come in the traditional 256-entry variety, which take
317 // up 768 bytes of RAM, and lightweight 16-entry varieties. The 16-entry
318 // variety automatically interpolates between its entries to produce
319 // a full 256-element color map, but at a cost of only 48 bytes or RAM.
320 //
321 // Basic operation is like this: (example shows the 16-entry variety)
322 // 1. Declare your palette storage:
323 // CRGBPalette16 myPalette;
324 //
325 // 2. Fill myPalette with your own 16 colors, or with a preset color scheme.
326 // You can specify your 16 colors a variety of ways:
327 // CRGBPalette16 myPalette(
328 // CRGB::Black,
329 // CRGB::Black,
330 // CRGB::Red,
331 // CRGB::Yellow,
332 // CRGB::Green,
333 // CRGB::Blue,
334 // CRGB::Purple,
335 // CRGB::Black,
336 //
337 // 0x100000,
338 // 0x200000,
339 // 0x400000,
340 // 0x800000,
341 //
342 // CHSV( 30,255,255),
343 // CHSV( 50,255,255),
344 // CHSV( 70,255,255),
345 // CHSV( 90,255,255)
346 // );
347 //
348 // Or you can initiaize your palette with a preset color scheme:
349 // myPalette = RainbowStripesColors_p;
350 //
351 // 3. Any time you want to set a pixel to a color from your palette, use
352 // "ColorFromPalette(...)" as shown:
353 //
354 // uint8_t index = /* any value 0..255 */;
355 // leds[i] = ColorFromPalette( myPalette, index);
356 //
357 // Even though your palette has only 16 explicily defined entries, you
358 // can use an 'index' from 0..255. The 16 explicit palette entries will
359 // be spread evenly across the 0..255 range, and the intermedate values
360 // will be RGB-interpolated between adjacent explicit entries.
361 //
362 // It's easier to use than it sounds.
363 //

848 // nblendPaletteTowardPalette:
849 // Alter one palette by making it slightly more like
850 // a 'target palette', used for palette cross-fades.
851 //
852 // It does this by comparing each of the R, G, and B channels
853 // of each entry in the current palette to the corresponding
854 // entry in the target palette and making small adjustments:
855 // If the Red channel is too low, it will be increased.
856 // If the Red channel is too high, it will be slightly reduced.
857 // ... and likewise for Green and Blue channels.
858 //
859 // Additionally, there are two significant visual improvements
860 // to this algorithm implemented here. First is this:
861 // When increasing a channel, it is stepped up by ONE.
862 // When decreasing a channel, it is stepped down by TWO.
863 // Due to the way the eye perceives light, and the way colors
864 // are represented in RGB, this produces a more uniform apparent
865 // brightness when cross-fading between most palette colors.
866 //
867 // The second visual tweak is limiting the number of changes
868 // that will be made to the palette at once. If all the palette
869 // entries are changed at once, it can give a muddled appearance.
870 // However, if only a few palette entries are changed at once,
871 // you get a visually smoother transition: in the middle of the
872 // cross-fade your current palette will actually contain some
873 // colors from the old palette, a few blended colors, and some
874 // colors from the new palette.
875 // The maximum number of possible palette changes per call
876 // is 48 (sixteen color entries time three channels each).
877 // The default 'maximim number of changes' here is 12, meaning
878 // that only approximately a quarter of the palette entries
879 // will be changed per call.

881  CRGBPalette16& targetPalette,
882  uint8_t maxChanges=24);
