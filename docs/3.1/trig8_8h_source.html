38 LIB8STATIC int16_t sin16_avr( uint16_t theta )
39 {
40  static const uint8_t data[] =
41  { 0, 0, 49, 0, 6393%256, 6393/256, 48, 0,
42  12539%256, 12539/256, 44, 0, 18204%256, 18204/256, 38, 0,
43  23170%256, 23170/256, 31, 0, 27245%256, 27245/256, 23, 0,
44  30273%256, 30273/256, 14, 0, 32137%256, 32137/256, 4 /*,0*/ };
45 
46  uint16_t offset = (theta & 0x3FFF);
47 
48  // AVR doesn't have a multi-bit shift instruction,
49  // so if we say "offset >>= 3", gcc makes a tiny loop.
50  // Inserting empty volatile statements between each
51  // bit shift forces gcc to unroll the loop.
52  offset >>= 1; // 0..8191
53  asm volatile("");
54  offset >>= 1; // 0..4095
55  asm volatile("");
56  offset >>= 1; // 0..2047
57 
58  if( theta & 0x4000 ) offset = 2047 - offset;
59 
60  uint8_t sectionX4;
61  sectionX4 = offset / 256;
62  sectionX4 *= 4;
63 
64  uint8_t m;
65 
66  union {
67  uint16_t b;
68  struct {
69  uint8_t blo;
70  uint8_t bhi;
71  };
72  } u;
73 
74  //in effect u.b = blo + (256 * bhi);
75  u.blo = data[ sectionX4 ];
76  u.bhi = data[ sectionX4 + 1];
77  m = data[ sectionX4 + 2];
78 
79  uint8_t secoffset8 = (uint8_t)(offset) / 2;
80 
81  uint16_t mx = m * secoffset8;
82 
83  int16_t y = mx + u.b;
84  if( theta & 0x8000 ) y = -y;
85 
86  return y;
87 }

168 {

172  "sbrc %[theta],6 \n\t"
173  "com %[offset] \n\t"
174  : [theta] "+r" (theta), [offset] "+r" (offset)
175  );
176 
177  offset &= 0x3F; // 0..63
178 
179  uint8_t secoffset = offset & 0x0F; // 0..15
180  if( theta & 0x40) secoffset++;
181 
182  uint8_t m16; uint8_t b;
183 
184  uint8_t section = offset >> 4; // 0..3
185  uint8_t s2 = section * 2;
186 
187  const uint8_t* p = b_m16_interleave;
188  p += s2;
189  b = *p;
190  p++;
191  m16 = *p;

193  uint8_t mx;

196  "mul %[m16],%[secoffset] \n\t"

204  "or %[mx], %[xr1] \n\t"
