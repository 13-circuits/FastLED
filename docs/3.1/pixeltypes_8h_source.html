473  // setParity adjusts the color in the smallest
474  // way possible so that the parity of the color
475  // is now the desired value. This allows you to
476  // 'hide' one bit of information in the color.
477  //
478  // Ideally, we find one color channel which already
479  // has data in it, and modify just that channel by one.
480  // We don't want to light up a channel that's black
481  // if we can avoid it, and if the pixel is 'grayscale',
482  // (meaning that R==G==B), we modify all three channels
483  // at once, to preserve the neutral hue.
484  //
485  // There's no such thing as a free lunch; in many cases
486  // this 'hidden bit' may actually be visible, but this
487  // code makes reasonable efforts to hide it as much
488  // as is reasonably possible.
489  //
490  // Also, an effort is made to have make it such that
491  // repeatedly setting the parity to different values
492  // will not cause the color to 'drift'. Toggling
493  // the parity twice should generally result in the
494  // original color again.
495  //
496  inline void setParity( uint8_t parity)
497  {
498  uint8_t curparity = getParity();
499 
500  if( parity == curparity) return;
501 
502  if( parity ) {
503  // going 'up'
504  if( (b > 0) && (b < 255)) {
505  if( r == g && g == b) {
506  r++;
507  g++;
508  }
509  b++;
510  } else if( (r > 0) && (r < 255)) {
511  r++;
512  } else if( (g > 0) && (g < 255)) {
513  g++;
514  } else {
515  if( r == g && g == b) {
516  r ^= 0x01;
517  g ^= 0x01;
518  }
519  b ^= 0x01;
520  }
521  } else {
522  // going 'down'
523  if( b > 1) {
524  if( r == g && g == b) {
525  r--;
526  g--;
527  }
528  b--;
529  } else if( g > 1) {
530  g--;
531  } else if( r > 1) {
532  r--;
533  } else {
534  if( r == g && g == b) {
535  r ^= 0x01;
536  g ^= 0x01;
537  }
538  b ^= 0x01;
539  }
540  }
